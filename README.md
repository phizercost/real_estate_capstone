const MNEMONIC = "xxxxxxx xxxxxxxx xxxxxxx xxxxxxx xxxxxxx xxxxxxxx xxxxxxx xxxxxx xxxxxxx xxxxxx xxxxx xxxxxxx";
module.exports = {
    INFURA_KEY,
    MNEMONIC
};
